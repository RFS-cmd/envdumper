local Create = function(Class, Properties)
    local Object = Instance.new(Class)

    if Properties then
        for Key, Value in Properties do
           Object[Key] = Value 
        end
    end

    return Object

local Module = {}
Module.__index = Module

function Module:Drag(Data)
	assert(Data["Interface"], "Missing interface")

	Data = setmetatable(Data, Module)

	Data.CanDrag = Data.CanDrag or true
	Data.Speed = Data.Speed or 0

	Data.Connections = {}

	local Dragging, Start, Position = false, Vector2.zero, UDim2.new()

	table.insert(Data.Connections, Data.Interface.InputBegan:Connect(function(Input)
		if (Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch) and not UserInputService:GetFocusedTextBox() then
			Dragging = true
			Start = Vector2.new(Input.Position.X, Input.Position.Y)
			Position = Data.Interface.Position

			Data.Connections.InputChanged = Input.Changed:Connect(function()
				if Input.UserInputState == Enum.UserInputState.End then
					Dragging = false

					Data.Connections.InputChanged:Disconnect()
				end
			end)
		end
	end))

	table.insert(Data.Connections, UserInputService.InputChanged:Connect(function(Input)
		if Data.CanDrag and Dragging and (Input.UserInputType == Enum.UserInputType.MouseMovement or Input.UserInputType == Enum.UserInputType.Touch) then
			local Delta = Vector2.new(Input.Position.X, Input.Position.Y) - Start
			local DragPosition = UDim2.new(
				Position.X.Scale,     Position.X.Offset + Delta.X,
				Position.Y.Scale,     Position.Y.Offset + Delta.Y
			)

			if Data.Speed > 0 then
				TweenService:Create(Data.Interface, TweenInfo.new(Data.Speed), {
					Position = DragPosition
				}):Play()
			else
				Data.Interface.Position = DragPosition
			end
		end
	end))
	
	return Data
end

local function ApplyLiquidGlass(Frame, Connections)
	-- Base glass appearance
	Frame.BackgroundColor3 = Color3.fromRGB(200, 210, 230)
	Frame.BackgroundTransparency = 0.72

	-- Rounded corners (preserve existing UICorner if present)
	if not Frame:FindFirstChildOfClass("UICorner") then
		local Corner = Instance.new("UICorner")
		Corner.CornerRadius = UDim.new(0, 6)
		Corner.Parent = Frame
	end

	-- Rim / border
	local Stroke = Instance.new("UIStroke")
	Stroke.Thickness = 1
	Stroke.Color = Color3.fromRGB(255, 255, 255)
	Stroke.Transparency = 0.55
	Stroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
	Stroke.Parent = Frame

	-- Specular highlight gradient (bright top → subtle bottom)
	local Gradient = Instance.new("UIGradient")
	Gradient.Rotation = 90
	Gradient.Color = ColorSequence.new({
		ColorSequenceKeypoint.new(0,   Color3.fromRGB(255, 255, 255)),
		ColorSequenceKeypoint.new(0.4, Color3.fromRGB(210, 220, 240)),
		ColorSequenceKeypoint.new(1,   Color3.fromRGB(160, 175, 205)),
	})
	Gradient.Transparency = NumberSequence.new({
		NumberSequenceKeypoint.new(0,   0.55),
		NumberSequenceKeypoint.new(0.5, 0.78),
		NumberSequenceKeypoint.new(1,   0.85),
	})
	Gradient.Parent = Frame

	-- Subtle shimmer: oscillate gradient rotation slightly
	local shimmerAngle = 90
	local shimmerDir   = 1
	local ShimmerConn  = RunService.RenderStepped:Connect(function(dt)
		shimmerAngle = shimmerAngle + dt * 8 * shimmerDir
		if shimmerAngle >= 96 then shimmerDir = -1
		elseif shimmerAngle <= 84 then shimmerDir = 1 end
		Gradient.Rotation = shimmerAngle
	end)
	if Connections then
		table.insert(Connections, ShimmerConn)
	end

	-- Hover: pop up (size grows slightly, glass brightens, rim sharpens)
	local baseSize = Frame.Size
	Frame.MouseEnter:Connect(function()
		TweenService:Create(Frame, TweenInfo.new(0.18, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
			BackgroundTransparency = 0.55,
			Size = UDim2.new(baseSize.X.Scale, baseSize.X.Offset, baseSize.Y.Scale, baseSize.Y.Offset + 2),
		}):Play()
		TweenService:Create(Stroke, TweenInfo.new(0.18), {
			Transparency = 0.25,
		}):Play()
	end)

	Frame.MouseLeave:Connect(function()
		TweenService:Create(Frame, TweenInfo.new(0.22, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
			BackgroundTransparency = 0.72,
			Size = baseSize,
		}):Play()
		TweenService:Create(Stroke, TweenInfo.new(0.22), {
			Transparency = 0.55,
		}):Play()
	end)

	return Stroke, Gradient
end

function Class(Configuration: table, Class: table)
	return setmetatable(Configuration, {__index = Class})
end

function Bounds(Left, Top, Right, Bottom)
	return {
		Left = Left,
		Top = Top,
		Right = Right,
		Bottom = Bottom
	}
end

function GetBounds(GuiObject: GuiObject)
	return Bounds(
		GuiObject.AbsolutePosition.X,
		GuiObject.AbsolutePosition.Y,
		GuiObject.AbsolutePosition.X + GuiObject.AbsoluteSize.X,
		GuiObject.AbsolutePosition.Y + GuiObject.AbsoluteSize.Y
	)
end

function InBounds(Position: Vector2, Bounds: table)
	return Position.X >= Bounds.Left and Position.X <= Bounds.Right
		and Position.Y >= Bounds.Top and Position.Y <= Bounds.Bottom
end

function Window:Setup()
	self.Interface.Parent = self.Parent
	self.Interface.Main.Title.Text = self.Name

	self.Interface.Main.Close.Image = GetCustomAsset("biggiehub/assets/close.png")
	self.Interface.Main.Maximize.Image = GetCustomAsset("biggiehub/assets/maximize.png")
	self.Interface.Main.Minimize.Image = GetCustomAsset("biggiehub/assets/minimize.png")
	self.Interface.Main.Content.SearchBar.Search.Image = GetCustomAsset("biggiehub/assets/search.png")

	self.Interface.DisplayOrder = math.huge

	if self.AutoShow then
		self:Set(true)
	end

	if self.MobileButton then
		table.insert(self.Connections, self.MobileButton.MouseButton1Click:Connect(function()
			self:Set(not self.Open)
		end))
	end

	table.insert(self.Connections, UserInputService.InputBegan:Connect(function(Input, ProcessedEvent)
		if Input.KeyCode == self.KeyCode then
			self:Set(not self.Open)
		end
	end))

	table.insert(self.Connections, self.Interface.Main.Close.MouseButton1Click:Connect(function()
		self:Set(false)
	end))
	
	table.insert(self.Connections, self.Interface.Main.Maximize.MouseButton1Click:Connect(function()
		self:SetMaximized(not self.Maximized)
	end))
	
	table.insert(self.Connections, self.Interface.Main.Minimize.MouseButton1Click:Connect(function()
		self:Set(false)
	end))
	
	if not self.MaximizeEnabled then
		TweenService:Create(self.Interface.Main.Maximize, TweenInfo.new(0.1), {
			["ImageColor3"] = CONSTANTS.MAXIMIZE_DISABLED_COLOR
		}):Play()
	end

	self.Drag = Module:Drag({
		["Interface"] = self.Interface.Main,
		["Speed"] = 0,
		["CanDrag"] = true
	})
	
	table.insert(self.Connections, self.Interface.Main.Content.SearchBar.TextBox:GetPropertyChangedSignal("Text"):Connect(function()
		local Text = self.Interface.Main.Content.SearchBar.TextBox.Text

		for _, Tab in self.Tabs do
			if Tab.Interface:FindFirstChild("Content") == nil then
				continue
			end
			
			for _, Object in Tab.Interface.Content:GetChildren() do
				if not Object:IsA("GuiObject") then continue end
				
				if string.len(Text) > 0 and not string.match(string.lower(Object.Name), string.lower(Text)) then
					Object.Visible = false
				else
					Object.Visible = true
				end
			end
		end
	end))

	self.Snowflakes = {}

	-- Single unified background for the entire window, sits behind all tabs and content
	local bgImage
	if getcustomasset then
		local ok, result = pcall(getcustomasset, "rbxassetid://87906311514772")
		bgImage = (ok and result and result ~= "") and result or "rbxassetid://87906311514772"
	else
		bgImage = "rbxassetid://87906311514772"
	end

	local Background = Instance.new("ImageLabel")
	Background.Name = "__Background"
	Background.BackgroundTransparency = 1
	Background.Image = bgImage
	Background.Size = UDim2.new(1, 0, 1, 0)
	Background.Position = UDim2.new(0, 0, 0, 0)
	Background.ZIndex = 0
	Background.ScaleType = Enum.ScaleType.Crop
	Background.Parent = self.Interface.Main

	for _, Child in self.Interface.Main:GetChildren() do
		if Child ~= Background and Child:IsA("GuiObject") then
			Child.ZIndex = math.max(Child.ZIndex, 1)
		end
	end

	self.Background = Background
end

function Window:ClearSnow()

		Snowflake:Destroy()

function Window:SetSnowEffect(Mode: boolean)
	if self.Connections.SnowEffect then
		self.Connections.SnowEffect:Disconnect()
		self.Connections.SnowEffect = nil
	end

	if Mode then
		self:ClearSnow()

		local LastSnow = tick()
		self.Connections.SnowEffect = RunService.RenderStepped:Connect(function()
			if (tick() - LastSnow) > 0.05 then
				LastSnow = tick()

				local Size = math.random(18, 20) / 10

				local Snowflake = Instance.new("Frame")
				Snowflake.Size = UDim2.new(0, Size, 0, Size)
				Snowflake.BackgroundColor3 = Color3.new(1, 1, 1)
				Snowflake.Position = UDim2.new(0, math.random(0, self.Interface.Main.AbsoluteSize.X), 0, -5)
				Snowflake.BackgroundTransparency = math.random(0, 9) / 10
				Snowflake.Parent = self.Interface.Main

				local UICorner = Instance.new("UICorner")
				UICorner.Parent = Snowflake
				UICorner.CornerRadius = UDim.new(1, 0)

				table.insert(self.Snowflakes, Snowflake)

				local Tween = TweenService:Create(Snowflake, TweenInfo.new(math.random(20, 30) / 10, Enum.EasingStyle.Linear), {
					["Position"] = UDim2.new(0, Snowflake.Position.X.Offset, 1, 5)
				})

				Tween:Play()
				Tween.Completed:Connect(function()
					Snowflake:Destroy()
				end)
			end
		end)
	else
		self:ClearSnow()
	end
end

			["BackgroundColor3"] = CONSTANTS.TAB_COLOR
		}):Play()
	end))

function Window:Save()
	if not self.Initialized then
		return
	end

	local Data = {}

	for Flag, Feature in self.Features do
		Data[Flag] = Feature:Get()
	end

	task.spawn(pcall, writefile, "biggiehub/config/" .. GAME_ID .. ".json", HttpService:JSONEncode(Data))
end

function Window:Load(Ignore)
    Ignore = Ignore or {}
	self.SaveFile = isfile("biggiehub/config/" .. GAME_ID .. ".json") and HttpService:JSONDecode(readfile("biggiehub/config/" .. GAME_ID .. ".json")) or {}

	for Flag, Data in self.SaveFile do
		if self.Features[Flag] and not table.find(Ignore, Flag) then
			self.Features[Flag]:Set(Data)
		end
	end

	self.Initialized = true
end

function Tab:CreateSlider(Data: SliderConfigurator)
	local Slider = Class(Data, Slider)

	Slider.Window = self.Window
	Slider.Tab = self

	Slider.Interface = self.Window.Interface.Examples.Slider:Clone()
	Slider.Mainframe = self.Window.Interface.Examples.Mainframe:Clone()

	Slider.Connections = {}
	Slider.Status = (Slider.Status == nil and true) or Slider.Status
	
	Slider.Range = (Slider.Range == nil and {0, 10}) or Slider.Range
	Slider.Increment = (Slider.Range == nil and 1) or Slider.Increment

	--Slider.Value = Slider.Flag and (self.Window.Objects[Slider.Flag] ~= nil and self.Window.Objects[Slider.Flag]) or Slider.Value
	
	Slider.Loop = nil;

	if math.min((Slider.Range[2] - Slider.Range[1]) % Slider.Increment, Slider.Increment - ((Slider.Range[2] - Slider.Range[1]) % Slider.Increment)) > 0.001 then
		Slider:Flush()
		
		error("Slider range is not divisible by increment")
	end
	
	if math.min(Slider.Value % Slider.Increment, Slider.Increment - (Slider.Value % Slider.Increment)) > 0.001 then
		Slider:Flush()

		error("Slider value is not divisible by increment")
	end

	Slider:Setup()

	self.Window.Features[Slider.Flag] = Slider

	return Slider
end

function Slider:Setup()
	self.Interface.Text.Text = self.Name
	self.Interface.Name = self.Name

	self.Interface.Visible = true
	self.Interface.Parent = self.Mainframe

	self.Mainframe.Parent = self.Tab.Interface.Content
	self.Mainframe.Name = self.Interface.Name
	
	self.Mainframe.Visible = true
	ApplyLiquidGlass(self.Mainframe, self.Connections)
	
	local Dragging = false
	table.insert(self.Connections, self.Interface.Drag.InputBegan:Connect(function(Input, Processed)
		if not self.Status then return end
		
		if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
			self.Interface.Slider.Size = UDim2.fromOffset(200, 8)
			
			Dragging = true
			
			self.Window.Drag.CanDrag = false
			self.Loop = RunService.RenderStepped:Connect(function()
				if not self.Status then return end
				
				if Dragging then
					self:Step();
				end
			end)
			
			table.insert(self.Connections, self.Loop)
		end
	end))
	
	table.insert(self.Connections, self.Interface.Drag.InputEnded:Connect(function(Input)
		if not self.Status then return end
		
		if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch and Dragging then
			self.Interface.Slider.Size = UDim2.fromOffset(200, 6)
			
			Dragging = false
			
			self.Window.Drag.CanDrag = true
			
			if self.Loop and self.Disconnect then
				self.Loop:Disconnect()
				self.Loop = nil
			end
		end
	end))
	
	self:Update(true)
end

		["BackgroundColor3"] = self.Status and CONSTANTS.SLIDER_COLOR or CONSTANTS.DISABLED_SLIDER_COLOR

		}):Play()
	else
		self.Interface.Slider.Value.Size = UDim2.new(0, self:CalculatePosition(self:Get()), 1, 0)
	end

function Slider:CalculateValue(X: number)
	local Minimum = self.Range[1]
	local Maximum = self.Range[2]
	local Increment = self.Increment

	local Gaps = Maximum - Minimum
	local Gap = self.Interface.Slider.AbsoluteSize.X / Gaps

	local Raw = Minimum + (X / Gap)
	local Closest = math.floor((Raw - Minimum) / Increment + 0.5) * Increment + Minimum

	local IncrementStr = string.format("%.10f", Increment):gsub("0+$", "")
	local Decimals = IncrementStr:match("%.(%d*)$")
	Decimals = Decimals and #Decimals or 0
	local Rounded = tonumber(string.format("%." .. Decimals .. "f", Closest))

	return math.clamp(Rounded, Minimum, Maximum)
end


function Slider:CalculatePosition(Value: number)
	local Minimum = self.Range[1]
	local Maximum = self.Range[2]

	local Gaps = Maximum - Minimum
	local Gap = self.Interface.Slider.AbsoluteSize.X / Gaps

	Value = math.clamp(Value, Minimum, Maximum)

	local SliderPosition = (Value - Minimum) * Gap

	return SliderPosition
end

function Slider:Get()
	return self.Value
end

function Slider:Set(Value)
	local OldValue = self:Get();
	
	self.Value = Value
	self:Update()

	if self.Window.AutoSave then
		self.Window:Save()
	end
	
	if self.FireOnStepped or (not (self.FireOnStepped) and self:Get() ~= OldValue) then
		task.spawn(self.Callback, self:Get())
	end
end

function Slider:Step()
	local SliderPosition = Mouse.X - self.Interface.Slider.AbsolutePosition.X;
	local Value = self:CalculateValue(SliderPosition);
	
	self:Set(Value)
end

function Dropdown:Select(Options: table)
	for Index, Option in Options do
		if not table.find(self.Options, Option) then
			continue
		end
		
		if self:IsSelected(Option) then
			table.remove(self.Selected, self:IsSelected(Option))
		end
		
		table.insert(self.Selected, Option)
	end
	
	if not self.Multiselect then
		while #self.Selected > 1 do
			table.remove(self.Selected, 1)
		end
	end
	
	self:Update()
    task.spawn(self.Callback, self.Selected)
end

	for Index, Option in Options do
		if not table.find(self.Options, Option) then
			continue
		end

		if self:IsSelected(Option) then
			table.remove(self.Selected, self:IsSelected(Option))
		else
			table.insert(self.Selected, Option)
		end
	end
	
	if not self.Multiselect then
		while #self.Selected > 1 do
			table.remove(self.Selected, 1)
		end
	end
	
	self:Update()
	task.spawn(self.Callback, self.Selected)
end

		if self.Loop then
			self.Loop:Disconnect()
			self.Loop = nil
		end

			self.Picker:Destroy()
			self.Picker = nil

			if self.Dragging == CONSTANTS.DRAGGING_SATURATION_BRIGHTNESS then
				local Bounds = GetBounds(self.Picker.SelectionFrame)

				local Bounds = GetBounds(self.Picker.Hue)
				local MouseX = math.clamp(Mouse.X - Bounds.Left, 0, Bounds.Right - Bounds.Left)
				
				local Hue = MouseX / (Bounds.Right - Bounds.Left)

				self.Value.Hue = Hue
				self:Update()

				task.spawn(self.Callback, self:Get())
			end
		end)

		table.insert(self.Connections, self.Loop)

			if Input.UserInputType == Enum.UserInputType.MouseButton1 then

			end

			end
		end))

		Debounce = false
	end))

	self:Update()

function Picker:Update(Instant: boolean?)
	local Color = self:Get()
	self.Interface.Color.Text = `R: {math.floor(Color.R * 255)}  G: {math.floor(Color.G * 255)}  B: {math.floor(Color.B * 255)}`
	
	TweenService:Create(self.Interface.Text, TweenInfo.new(0.1), {
		["TextColor3"] = self.Status and CONSTANTS.TEXT_COLOR or CONSTANTS.DISABLED_TEXT_COLOR
	}):Play()
	
	TweenService:Create(self.Interface.Color, TweenInfo.new(0.1), {
		["TextColor3"] = self.Status and CONSTANTS.TEXT_COLOR or CONSTANTS.DISABLED_TEXT_COLOR
	}):Play()
	
	TweenService:Create(self.Interface.Picker, TweenInfo.new(Instant and 0 or 0.1), {
		["BackgroundColor3"] = self:Get()
	}):Play()
	
	if self.Picker then
		local Hue, Saturation, Brightness = self.Value.Hue, self.Value.Saturation, self.Value.Brightness
		local X, Y, HueX = self:CalculatePosition(Saturation, Brightness, Hue)
		
		TweenService:Create(self.Picker.SelectionFrame.Selection, TweenInfo.new(Instant and 0 or 0.1), {
			["Position"] = UDim2.new(0, X, 0, Y)
		}):Play()
		
		TweenService:Create(self.Picker.Hue.Selection, TweenInfo.new(Instant and 0 or 0.1), {
			["Position"] = UDim2.new(0, HueX, 0.5, 0)
		}):Play()

		TweenService:Create(self.Picker.Box.Color, TweenInfo.new(Instant and 0 or 0.1), {
			["BackgroundColor3"] = self:Get()
		}):Play()
		
		self.Picker.Box.Saturation.Value.Color = ColorSequence.new({ColorSequenceKeypoint.new(0, Color3.new(1, 1, 1)), ColorSequenceKeypoint.new(1, Color3.fromHSV(self.Value.Hue, 1, 1))})
	end
end

function Picker:CalculateColor(ColorX, ColorY)
	local Saturation = ((1 / self.Picker.SelectionFrame.Size.X.Offset) * ColorX)
	local Brightness = 1 - ((1 / self.Picker.SelectionFrame.Size.Y.Offset) * ColorY)
	
	return Saturation, Brightness
end

function Picker:CalculatePosition(Saturation, Brightness, Hue)
	local Bounds = GetBounds(self.Picker.Hue)
	
	local ColorX = Saturation * self.Picker.SelectionFrame.Size.X.Offset
	local ColorY = (1 - Brightness) * self.Picker.SelectionFrame.Size.Y.Offset
	
	local HueX = Hue * (Bounds.Right - Bounds.Left)

	return ColorX, ColorY, HueX
end

function Picker:SetStatus(Status: boolean)
	self.Status = Status
	self:Update()
end

function Picker:Get()
	return Color3.fromHSV(self.Value.Hue, self.Value.Saturation, self.Value.Brightness)
end

function Picker:Set(Value: Color3)
	local Hue, Saturation, Brightness = Value:ToHSV()
	
	self.Value = {
		Hue = Hue,
		Saturation = Saturation,
		Brightness = Brightness
	};
	
	self:Update()
end

end
